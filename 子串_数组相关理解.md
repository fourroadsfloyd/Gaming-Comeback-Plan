## lc.560 和为k的子数组

```cpp
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int, int> cache;
        cache[0] = 1;

        int pre = 0, count = 0;
        for(int& x : nums)
        {
            pre += x;
            if(cache.count(pre - k))
            {
                count += cache[pre - k];
            }

            ++cache[pre];
        }

        return count;
    }
};
```

前缀和是指累加到每个元素时的值

需要注意前缀和容器中需要将初始值设为0，即num[0] - nums[-1] = nums[0]，保证第一个元素也满足前缀和公式

## lc.239 滑动窗口最大值

```cpp
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int n = nums.size();
        deque<int> q;
        for(int i=0;i<k;++i)
        {
            while(q.empty() == false && nums[i] >= nums[q.back()])
            {
                q.pop_back();
            }
            
            q.push_back(i);
        }

        vector<int> ans{nums[q.front()]};

        for(int i=k;i<n;++i)
        {
            while(q.empty() == false && nums[i] >= nums[q.back()])
            {
                q.pop_back();
            }
            
            q.push_back(i);

            while(q.front() <= i-k)
            {
                q.pop_front();
            }

            ans.push_back(nums[q.front()]);
        }

        return ans;
    }
};
```

单调队列的思想：保证队列是严格递减的！！当未入队元素大于等于队列末尾索引指向的元素时，将队列末尾弹出，接着将该元素索引塞进队列，这个操作保证了队列中保存的索引指向的元素严格递减

```cpp
while(q.empty() == false && nums[i] >= nums[q.back()])
{
    q.pop_back();
}
```

需要注意：**队列中存储的是元素索引**，这是为了pop出不在滑动窗口内的索引

```cpp
while(q.front() <= i-k)
{
    q.pop_front();
}
```

**这个还得多看，多理解单调队列的思想**



## lc.53 最大子数组和

```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int pre = 0;
        int ans = nums[0];
        
        for(int& x : nums)
        {
            pre = std::max(pre+x, x);
            ans = std::max(ans, pre);
        }

        return ans;
    }
};
```

这道题是一个简单的前缀和问题

需要注意**pre应该在比前面的值都大时才更新，且ans应该初值为nums[0]**



## lc.56 合并区间

```cpp
class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        vector<vector<int>> ans{};

        sort(intervals.begin(), intervals.end());

        for(int i=0;i<intervals.size();++i)
        {
            int L = intervals[i][0];
            int R = intervals[i][1];

            if(ans.size() == 0 || ans.back()[1] < L)
            {
                ans.push_back({L, R});
            }
            else
            {
                ans.back()[1] = std::max(ans.back()[1], R);
            }
        }

        return ans;
    }
};
```

sort函数对intervals进行排序，按照每个vector的首元素从小到大排序，当首元素相等时按照第二个元素排序

**何时塞进ans中？**

当ans为空时，直接push_back

当ans的末尾vector的末尾元素小于遍历的当前vector的首元素时，说明当前vector是一个新的区间，直接push_back

**最后更新区间的右端点**

```cpp
ans.back()[1] = std::max(ans.back()[1], R);
```

**这道题关键在于何时比较R**



## lc.189 轮转数组

```cpp
class Solution {
public:
    void reverse(vector<int>& nums, int l, int r)
    {
        while(l < r)
        {
            std::swap(nums[l], nums[r]);
            ++l;
            --r;
        }
    }

    void rotate(vector<int>& nums, int k) {
        int n = nums.size();
        k = k % n;

        reverse(nums, 0, n-1);
        reverse(nums, 0, k-1);
        reverse(nums, k, n-1);
    }
};
```

数组反转的操作需要理解，仅是一个双指针操作

**轮转数推导出从何时开始反转！**



## lc.238 除了自身以外的乘积

```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        vector<int> ans(nums.size());

        ans[0] = 1; //这里需要塞入数据，所以需要提前分配容量

        for(int i=1;i<nums.size();++i)
        {
            ans[i] = ans[i-1] * nums[i-1];
        }

        int R = 1;
        for(int i=nums.size()-1;i>=0;--i)
        {
            ans[i] = ans[i] * R;
            R *= nums[i];
        }

        return ans;
    }
};
```

第一次遍历，计算当前值左边的乘积，第二次遍历，根据当前值右侧的乘积以及第一次遍历的结果，计算最终结果

**为什么起始是 1？**

因为第一轮里 `answer[i]` 定义为“`i` 左边所有数的乘积”。

- 对 `i=0` 来说，左边没有元素，是**空乘积**
- 空乘积按数学定义就是乘法单位元 `1`



## lc.41 缺失的第一个正数

```cpp
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n = nums.size();

        for(int i=0;i<n;++i)
        {
            while(nums[i] > 0 && nums[i] <= n && nums[nums[i]-1] != nums[i])
            {
                std::swap(nums[nums[i]-1], nums[i]);
            }
        }

        for(int i=0;i<n;++i)
        {
            if(nums[i] != i + 1)
                return i+1;
        }

        return n+1;
    }
};
```
